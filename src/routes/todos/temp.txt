    async function getData(id) {
        const docRef = doc(db, 'todo', `${id}`);
        const docSnap = await getDoc(docRef);
        return docSnap.data;
    }

    class Todo {
    constructor (task, created, completed ) {
        this.task = task;
        this.created = created;
        this.completed = completed;
    }
}
